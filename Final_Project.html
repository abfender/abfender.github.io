import pandas as pd
import matplotlib.pyplot as plt
#import pandas and matplotlib for graphing

df = pd.read_csv("../finalProject/health.csv", sep=",")
#read dataset into variable df

df.dropna(how = 'any', inplace = True)
#delete rows with any NaN vlaues to allow for cleaner graphing

#map each disease to a color to see prevalence of each overtime in a graph
cols = df["disease"].map({
  "MEASLES": "green",
  "POLIO": "yellow",
  "SMALLPOX": "red",
  "PERTUSSIS": "orange",
  "HEPATITIS A": "pink", 
  "RUBELLA": "purple",
  "MUMPS": "blue",
  
})

#plot number of cases and year on a scatterplot with color mapping
df.plot.scatter(x="year", y = "number", c = cols)
